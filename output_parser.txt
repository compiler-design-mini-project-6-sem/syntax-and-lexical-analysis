
GRAMMAR:
program -> BEGIN stmt_list END
stmt_list -> stmt stmt_list
stmt_list -> ε
stmt -> PRINT expr ;
stmt -> declaration
stmt -> assignment
stmt -> for_loop
declaration -> type var_list ;
type -> INTEGER
type -> REAL
type -> STRING
var_list -> IDENTIFIER var_list_tail
var_list_tail -> , IDENTIFIER var_list_tail
var_list_tail -> ε
assignment -> IDENTIFIER := expr ;
for_loop -> FOR IDENTIFIER := expr TO expr stmt_list END
expr -> IDENTIFIER
expr -> NUMBER
expr -> STRING

FIRST SETS:
+----------------+----------------------------------------+
| Non-Terminal   | First Set                              |
+================+========================================+
| program        | BEGIN                                  |
+----------------+----------------------------------------+
| stmt_list      | PRINT, INTEGER, REAL, STRING, FOR, END |
+----------------+----------------------------------------+
| declaration    | INTEGER, REAL, STRING                  |
+----------------+----------------------------------------+
| var_list       | IDENTIFIER                             |
+----------------+----------------------------------------+
| var_list_tail  | ,, ε                                   |
+----------------+----------------------------------------+

+----------------+----------------------------------------+
| for_loop       | FOR                                    |
+----------------+----------------------------------------+
| expr           | IDENTIFIER, NUMBER, STRING             |
+----------------+----------------------------------------+
| type           | INTEGER, REAL, STRING                  |
+----------------+----------------------------------------+
